# Generics

[← 返回数据类型](../)

泛型是 Go 1.18 引入的重要特性，允许编写能够处理多种类型的代码，同时保持类型安全。

<Badge text="专业级" type="warning" />

## 泛型基础

### 为什么需要泛型

在没有泛型之前，要实现通用的功能需要：

```go
// ❌ 为每种类型重复编写
func MinInt(a, b int) int {
    if a < b {
        return a
    }
    return b
}

func MinFloat64(a, b float64) float64 {
    if a < b {
        return a
    }
    return b
}

// ❌ 使用 interface{} 失去类型安全
func Min(a, b interface{}) interface{} {
    // 运行时类型检查，容易出错
}
```

使用泛型后：

```go
// ✅ 类型安全 + 代码复用
func Min[T constraints.Ordered](a, b T) T {
    if a < b {
        return a
    }
    return b
}

func main() {
    fmt.Println(Min(3, 5))           // int: 3
    fmt.Println(Min(3.14, 2.71))    // float64: 2.71
    fmt.Println(Min("a", "b"))      // string: "a"
}
```

### 类型参数

泛型函数在函数名后使用方括号声明类型参数：

```go
func 函数名[类型参数列表](参数列表) 返回值列表
```

```go
// 泛型函数
func First[T any](slice []T) T {
    if len(slice) == 0 {
        var zero T
        return zero
    }
    return slice[0]
}

func main() {
    nums := []int{1, 2, 3}
    fmt.Println(First(nums))  // 1

    strs := []string{"a", "b", "c"}
    fmt.Println(First(strs))  // "a"
}
```

<Callout type="info" title="类型参数命名">
  类型参数通常使用单个大写字母（如 T、K、V）表示，这是 Go 的惯例。

  - `T` - Type（类型）
  - `K` - Key（键）
  - `V` - Value（值）
  - `E` - Element（元素）
</Callout>

### 类型约束

类型约束限定了类型参数可以使用的类型范围：

```go
import "golang.org/x/exp/constraints"

// T 可以是任何有序类型
func Max[T constraints.Ordered](a, b T) T {
    if a > b {
        return a
    }
    return b
}

// T 可以是任何整数类型
func Double[T constraints.Integer](n T) T {
    return n * 2
}
```

## 泛型函数

### 基本语法

```go
package main

import (
    "fmt"
    "golang.org/x/exp/constraints"
)

// 泛型函数：返回两个值的最大值
func Max[T constraints.Ordered](a, b T) T {
    if a > b {
        return a
    }
    return b
}

// 泛型函数：切片中的最大值
func SliceMax[T constraints.Ordered](slice []T) T {
    if len(slice) == 0 {
        var zero T
        return zero
    }

    max := slice[0]
    for _, v := range slice[1:] {
        if v > max {
            max = v
        }
    }
    return max
}

func main() {
    // 使用类型推断
    fmt.Println(Max(3, 5))              // 5
    fmt.Println(Max(3.14, 2.71))        // 3.14
    fmt.Println(Max("hello", "world"))  // "world"

    // 显式指定类型
    fmt.Println(Max[int](10, 20))       // 20

    // 使用切片
    nums := []int{3, 1, 4, 1, 5, 9}
    fmt.Println(SliceMax(nums))         // 9

    floats := []float64{3.14, 2.71, 1.41}
    fmt.Println(SliceMax(floats))       // 3.14
}
```

## 泛型类型

### 泛型结构体

```go
// 泛型栈
type Stack[T any] struct {
    elements []T
}

func NewStack[T any]() *Stack[T] {
    return &Stack[T]{
        elements: make([]T, 0),
    }
}

func (s *Stack[T]) Push(v T) {
    s.elements = append(s.elements, v)
}

func (s *Stack[T]) Pop() (T, bool) {
    if len(s.elements) == 0 {
        var zero T
        return zero, false
    }

    idx := len(s.elements) - 1
    v := s.elements[idx]
    s.elements = s.elements[:idx]
    return v, true
}

func (s *Stack[T]) Peek() (T, bool) {
    if len(s.elements) == 0 {
        var zero T
        return zero, false
    }
    return s.elements[len(s.elements)-1], true
}

func (s *Stack[T]) IsEmpty() bool {
    return len(s.elements) == 0
}

func main() {
    // 整数栈
    intStack := NewStack[int]()
    intStack.Push(1)
    intStack.Push(2)
    intStack.Push(3)

    for !intStack.IsEmpty() {
        v, _ := intStack.Pop()
        fmt.Println(v)  // 3, 2, 1
    }

    // 字符串栈
    strStack := NewStack[string]()
    strStack.Push("hello")
    strStack.Push("world")

    v, _ := strStack.Pop()
    fmt.Println(v)  // "world"
}
```

### 泛型接口

```go
// 存储接口
type Storage[T any] interface {
    Save(key string, value T) error
    Load(key string) (T, error)
    Delete(key string) error
}

// 内存存储实现
type MemoryStorage[T any] struct {
    data map[string]T
}

func NewMemoryStorage[T any]() *MemoryStorage[T] {
    return &MemoryStorage[T]{
        data: make(map[string]T),
    }
}

func (m *MemoryStorage[T]) Save(key string, value T) error {
    m.data[key] = value
    return nil
}

func (m *MemoryStorage[T]) Load(key string) (T, error) {
    v, ok := m.data[key]
    if !ok {
        var zero T
        return zero, fmt.Errorf("key not found: %s", key)
    }
    return v, nil
}

func (m *MemoryStorage[T]) Delete(key string) error {
    delete(m.data, key)
    return nil
}

func main() {
    // 存储用户配置
    configStorage := NewMemoryStorage[map[string]interface{}]()
    configStorage.Save("app", map[string]interface{}{
        "debug": true,
        "port":  8080,
    })

    config, _ := configStorage.Load("app")
    fmt.Printf("%+v\n", config)  // map[debug:true port:8080]
}
```

## 类型约束

### 预定义约束

Go 提供了几个常用的预定义约束：

```go
// golang.org/x/exp/constraints 包

// 所有有序类型
type Ordered interface {
    Integer | Float | ~string
}

// 所有整数类型
type Integer interface {
    Signed | Unsigned
}

// 有符号整数
type Signed interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64
}

// 无符号整数
type Unsigned interface {
    ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 | ~uintptr
}

// 浮点类型
type Float interface {
    ~float32 | ~float64
}

// 所有数值类型
type Complex interface {
    ~complex64 | ~complex128
}
```

使用示例：

```go
import "golang.org/x/exp/constraints"

// 只接受整数类型
func Sum[T constraints.Integer](nums []T) T {
    var sum T
    for _, n := range nums {
        sum += n
    }
    return sum
}

// 只接受浮点类型
func Average[T constraints.Float](nums []T) T {
    if len(nums) == 0 {
        return 0
    }
    var sum T
    for _, n := range nums {
        sum += n
    }
    return sum / T(len(nums))
}
```

### 自定义约束

```go
// 自定义约束：类型必须有 String() 方法
type Stringer interface {
    String() string
}

func PrintString[T Stringer](v T) {
    fmt.Println(v.String())
}

// 自定义约束：类型必须是数值且支持特定操作
type Number interface {
    constraints.Integer | constraints.Float
}

func Add[T Number](a, b T) T {
    return a + b
}

// 复合约束
type SizedNumber interface {
    Number
    ~int64 | ~float64  // 进一步限定为 int64 或 float64
}
```

### comparable 约束

`comparable` 是 Go 内置的约束，表示支持 `==` 和 `!=` 操作的类型：

```go
// 检查切片中是否包含某个值
func Contains[T comparable](slice []T, target T) bool {
    for _, v := range slice {
        if v == target {
            return true
        }
    }
    return false
}

// 从切片中查找值的索引
func Index[T comparable](slice []T, target T) int {
    for i, v := range slice {
        if v == target {
            return i
        }
    }
    return -1
}

func main() {
    nums := []int{1, 2, 3, 4, 5}
    fmt.Println(Contains(nums, 3))     // true
    fmt.Println(Contains(nums, 10))    // false
    fmt.Println(Index(nums, 4))        // 3

    strs := []string{"a", "b", "c"}
    fmt.Println(Contains(strs, "b"))   // true
    fmt.Println(Index(strs, "d"))      // -1
}
```

### any 约束

`any` 是 `interface{}` 的别名，表示任意类型：

```go
func Print[T any](v T) {
    fmt.Println(v)
}

func First[T any](slice []T) (T, bool) {
    if len(slice) == 0 {
        var zero T
        return zero, false
    }
    return slice[0], true
}
```

### ~ 支持底层类型

使用 `~` 可以包含自定义类型：

```go
// 不使用 ~：只能用标准类型
func AddWithoutTilde[T int | float64](a, b T) T {
    return a + b
}

// 使用 ~：包含底层类型
func AddWithTilde[T int | float64](a, b T) T {
    return a + b
}

type MyInt int
type MyFloat64 float64

func main() {
    var a, b MyInt = 1, 2

    // ❌ MyInt 不在约束范围内
    // AddWithoutTilde(a, b)  // 编译错误

    // ✅ MyInt 的底层类型是 int，符合约束
    result := AddWithTilde(a, b)
    fmt.Println(result)  // 3
}
```

<Callout type="tip" title="~ 的作用">
  `~` 表示"底层类型是"的意思：

  - `int` - 只有 int 类型
  - `~int` - int 和所有底层类型为 int 的自定义类型

  这对于支持自定义类型非常有用。
</Callout>

## 实际应用场景

### 1. 泛型工具函数

```go
// 切片过滤
func Filter[T any](slice []T, predicate func(T) bool) []T {
    var result []T
    for _, v := range slice {
        if predicate(v) {
            result = append(result, v)
        }
    }
    return result
}

// 切片映射
func Map[T, U any](slice []T, mapper func(T) U) []U {
    result := make([]U, len(slice))
    for i, v := range slice {
        result[i] = mapper(v)
    }
    return result
}

// 切片归约
func Reduce[T, U any](slice []T, initial U, reducer func(U, T) U) U {
    result := initial
    for _, v := range slice {
        result = reducer(result, v)
    }
    return result
}

func main() {
    nums := []int{1, 2, 3, 4, 5}

    // 过滤偶数
    evens := Filter(nums, func(n int) bool {
        return n%2 == 0
    })
    fmt.Println(evens)  // [2, 4]

    // 映射为平方
    squares := Map(nums, func(n int) int {
        return n * n
    })
    fmt.Println(squares)  // [1, 4, 9, 16, 25]

    // 求和
    sum := Reduce(nums, 0, func(acc, n int) int {
        return acc + n
    })
    fmt.Println(sum)  // 15
}
```

### 2. 泛型缓存

```go
type Cache[K comparable, V any] struct {
    data map[K]V
    mu   sync.RWMutex
}

func NewCache[K comparable, V any]() *Cache[K, V] {
    return &Cache[K, V]{
        data: make(map[K]V),
    }
}

func (c *Cache[K, V]) Set(key K, value V) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.data[key] = value
}

func (c *Cache[K, V]) Get(key K) (V, bool) {
    c.mu.RLock()
    defer c.mu.RUnlock()
    v, ok := c.data[key]
    return v, ok
}

func (c *Cache[K, V]) Delete(key K) {
    c.mu.Lock()
    defer c.mu.Unlock()
    delete(c.data, key)
}

func main() {
    // 字符串键，整数值缓存
    cache := NewCache[string, int]()
    cache.Set("a", 1)
    cache.Set("b", 2)

    v, ok := cache.Get("a")
    fmt.Println(v, ok)  // 1 true

    // 整数键，字符串值缓存
    cache2 := NewCache[int, string]()
    cache2.Set(1, "one")
    cache2.Set(2, "two")

    v2, ok2 := cache2.Get(2)
    fmt.Println(v2, ok2)  // two true
}
```

### 3. 泛型数据结构

```go
// 泛型二叉树节点
type TreeNode[T any] struct {
    Value T
    Left  *TreeNode[T]
    Right *TreeNode[T]
}

// 泛型二叉搜索树
type BinarySearchTree[T constraints.Ordered] struct {
    root *TreeNode[T]
}

func NewBinarySearchTree[T constraints.Ordered]() *BinarySearchTree[T] {
    return &BinarySearchTree[T]{}
}

func (bst *BinarySearchTree[T]) Insert(value T) {
    bst.root = bst.insert(bst.root, value)
}

func (bst *BinarySearchTree[T]) insert(node *TreeNode[T], value T) *TreeNode[T] {
    if node == nil {
        return &TreeNode[T]{Value: value}
    }

    if value < node.Value {
        node.Left = bst.insert(node.Left, value)
    } else if value > node.Value {
        node.Right = bst.insert(node.Right, value)
    }

    return node
}

func (bst *BinarySearchTree[T]) InOrderTraversal() []T {
    var result []T
    bst.inOrder(bst.root, &result)
    return result
}

func (bst *BinarySearchTree[T]) inOrder(node *TreeNode[T], result *[]T) {
    if node == nil {
        return
    }
    bst.inOrder(node.Left, result)
    *result = append(*result, node.Value)
    bst.inOrder(node.Right, result)
}

func main() {
    bst := NewBinarySearchTree[int]()
    bst.Insert(5)
    bst.Insert(3)
    bst.Insert(7)
    bst.Insert(1)
    bst.Insert(9)

    result := bst.InOrderTraversal()
    fmt.Println(result)  // [1 3 5 7 9]
}
```

### 4. 泛型通道

```go
// 泛型通道管理器
type ChannelManager[T any] struct {
    ch chan T
}

func NewChannelManager[T any](size int) *ChannelManager[T] {
    return &ChannelManager[T]{
        ch: make(chan T, size),
    }
}

func (cm *ChannelManager[T]) Send(value T) {
    cm.ch <- value
}

func (cm *ChannelManager[T]) Receive() T {
    return <-cm.ch
}

func (cm *ChannelManager[T]) Close() {
    close(cm.ch)
}

func (cm *ChannelManager[T]) Range(handler func(T)) {
    for v := range cm.ch {
        handler(v)
    }
}

func main() {
    cm := NewChannelManager[int](10)

    // 生产者
    go func() {
        for i := 0; i < 5; i++ {
            cm.Send(i)
        }
        cm.Close()
    }()

    // 消费者
    cm.Range(func(v int) {
        fmt.Println("Received:", v)
    })
}
```

## 最佳实践

### 1. 适度使用泛型

<Callout type="warning" title="不要过度使用">
  泛型是强大的工具，但不是所有场景都需要：

  ✅ **适合使用泛型**：
  - 容器类型（栈、队列、树）
  - 工具函数（过滤、映射、排序）
  - 数据结构（链表、哈希表）

  ❌ **不适合使用泛型**：
  - 只有一种类型的实现
  - 简单的特定场景代码
  - 使代码变得复杂且收益不大
</Callout>

### 2. 类型约束选择

```go
// ✅ 使用最宽松的约束
func Print[T any](v T) {  // any 足够了
    fmt.Println(v)
}

// ✅ 根据需要选择合适的约束
func Min[T constraints.Ordered](a, b T) T {  // 需要比较
    if a < b {
        return a
    }
    return b
}

// ✅ 使用 comparable 需要比较相等性
func Contains[T comparable](slice []T, target T) bool {
    for _, v := range slice {
        if v == target {
            return true
        }
    }
    return false
}
```

### 3. 避免过度复杂

```go
// ❌ 过于复杂
func ComplexFunc[A, B, C, D any](
    a A, b B, c C, d D,
) map[A]map[B]struct {
    x C
    y D
} {
    // ...
}

// ✅ 简化设计
type Pair[C, D any] struct {
    x C
    y D
}

func SimplifiedFunc[A, B comparable, C, D any](
    a A, b B, c C, d D,
) map[A]Pair[C, D] {
    // ...
}
```

### 4. 性能考虑

<Callout type="info" title="泛型性能">
  Go 的泛型实现采用编译时类型特化，通常不会引入运行时开销：

  - 类型参数在编译时被具体类型替换
  - 没有装箱/拆箱开销
  - 和手写非泛型代码性能相当

  但需要注意：
  - 代码体积会增加（每种具体类型生成一份代码）
  - 过度使用可能导致编译时间增加
</Callout>

### 5. 可读性优先

```go
// ✅ 清晰的类型参数命名
type Map[K comparable, V any] struct {
    data map[K]V
}

// ❌ 不清晰的命名
type Map[A any, B any] struct {
    data map[A]B
}
```

## 常见陷阱

### 1. 类型参数不能直接使用方法

```go
// ❌ 错误：T 不一定有 String() 方法
func PrintString[T any](v T) {
    fmt.Println(v.String())  // 编译错误
}

// ✅ 正确：使用约束
type Stringer interface {
    String() string
}

func PrintString[T Stringer](v T) {
    fmt.Println(v.String())
}
```

### 2. 泛型类型实例化

```go
type Stack[T any] struct {
    elements []T
}

// ❌ 错误：必须指定所有类型参数
var s1 Stack  // 编译错误

// ✅ 正确：指定类型参数
var s2 Stack[int]
var s3 = NewStack[string]()
```

### 3. 类型推断限制

```go
func Pair[T any](a, b T) (T, T) {
    return a, b
}

// ❌ 错误：类型推断失败
a, b := Pair(1, "string")  // 编译错误

// ✅ 正确：显式指定类型或保持类型一致
a, b := Pair[int>(1, 2)
a, b := Pair(1, 2)
```

1. **实现泛型函数 `Last`**：返回切片的最后一个元素

<details>
<summary>查看答案</summary>

```go
package main

import "fmt"

// Last 返回切片的最后一个元素
func Last[T any](slice []T) (T, error) {
    var zero T
    if len(slice) == 0 {
        return zero, fmt.Errorf("切片为空")
    }
    return slice[len(slice)-1], nil
}

func main() {
    // 整数切片
    ints := []int{1, 2, 3, 4, 5}
    lastInt, err := Last(ints)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Println("最后一个整数:", lastInt)
    }

    // 字符串切片
    strs := []string{"a", "b", "c"}
    lastStr, err := Last(strs)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Println("最后一个字符串:", lastStr)
    }

    // 空切片
    empty := []int{}
    _, err = Last(empty)
    if err != nil {
        fmt.Println("空切片错误:", err)
    }
}
```

**解释**：展示了泛型函数的基本用法，使用 `[T any]` 定义泛型类型参数。
</details>

2. **实现泛型队列 `Queue[T]`**：支持 Enqueue 和 Dequeue 操作

<details>
<summary>查看答案</summary>

```go
package main

import "fmt"

// Queue 泛型队列
type Queue[T any] struct {
    items []T
}

// NewQueue 创建新队列
func NewQueue[T any]() *Queue[T] {
    return &Queue[T]{
        items: make([]T, 0),
    }
}

// Enqueue 入队
func (q *Queue[T]) Enqueue(item T) {
    q.items = append(q.items, item)
}

// Dequeue 出队
func (q *Queue[T]) Dequeue() (T, error) {
    var zero T
    if len(q.items) == 0 {
        return zero, fmt.Errorf("队列为空")
    }

    item := q.items[0]
    q.items = q.items[1:]
    return item, nil
}

// Peek 查看队首元素
func (q *Queue[T]) Peek() (T, error) {
    var zero T
    if len(q.items) == 0 {
        return zero, fmt.Errorf("队列为空")
    }
    return q.items[0], nil
}

// IsEmpty 判断队列是否为空
func (q *Queue[T]) IsEmpty() bool {
    return len(q.items) == 0
}

// Size 返回队列大小
func (q *Queue[T]) Size() int {
    return len(q.items)
}

func main() {
    // 整数队列
    intQueue := NewQueue[int]()

    intQueue.Enqueue(1)
    intQueue.Enqueue(2)
    intQueue.Enqueue(3)

    fmt.Printf("队列大小: %d\n", intQueue.Size())

    for !intQueue.IsEmpty() {
        item, _ := intQueue.Dequeue()
        fmt.Println("出队:", item)
    }

    // 字符串队列
    strQueue := NewQueue[string]()
    strQueue.Enqueue("first")
    strQueue.Enqueue("second")

    item, _ := strQueue.Dequeue()
    fmt.Println("字符串出队:", item)
}
```

**解释**：展示了如何定义泛型类型，并在方法中使用类型参数。
</details>

3. **实现泛型哈希表 `HashMap[K, V]`**：支持基本的 CRUD 操作

<details>
<summary>查看答案</summary>

```go
package main

import (
    "fmt"
)

// HashMap 泛型哈希表（K 必须是可比较的类型）
type HashMap[K comparable, V any] struct {
    data map[K]V
}

// NewHashMap 创建新哈希表
func NewHashMap[K comparable, V any]() *HashMap[K, V] {
    return &HashMap[K, V]{
        data: make(map[K]V),
    }
}

// Put 添加或更新键值对
func (hm *HashMap[K, V]) Put(key K, value V) {
    hm.data[key] = value
}

// Get 获取值
func (hm *HashMap[K, V]) Get(key K) (V, bool) {
    value, ok := hm.data[key]
    return value, ok
}

// Delete 删除键
func (hm *HashMap[K, V]) Delete(key K) {
    delete(hm.data, key)
}

// ContainsKey 检查键是否存在
func (hm *HashMap[K, V]) ContainsKey(key K) bool {
    _, ok := hm.data[key]
    return ok
}

// Size 返回哈希表大小
func (hm *HashMap[K, V]) Size() int {
    return len(hm.data)
}

// Keys 返回所有键
func (hm *HashMap[K, V]) Keys() []K {
    keys := make([]K, 0, len(hm.data))
    for k := range hm.data {
        keys = append(keys, k)
    }
    return keys
}

// Values 返回所有值
func (hm *HashMap[K, V]) Values() []V {
    values := make([]V, 0, len(hm.data))
    for _, v := range hm.data {
        values = append(values, v)
    }
    return values
}

func main() {
    // 字符串到整数的映射
    ages := NewHashMap[string, int]()
    ages.Put("Alice", 30)
    ages.Put("Bob", 25)
    ages.Put("Charlie", 35)

    fmt.Println("=== 年龄映射 ===")
    if age, ok := ages.Get("Bob"); ok {
        fmt.Printf("Bob 的年龄: %d\n", age)
    }

    fmt.Printf("所有键: %v\n", ages.Keys())
    fmt.Printf("所有值: %v\n", ages.Values())

    ages.Delete("Charlie")
    fmt.Printf("删除后大小: %d\n", ages.Size())

    // 整数到字符串的映射
    names := NewHashMap[int, string]()
    names.Put(1, "one")
    names.Put(2, "two")
    names.Put(3, "three")

    fmt.Println("\n=== 数字映射 ===")
    fmt.Printf("2 的英文: %v\n", names.Get(2))
}
```

**解释**：展示了如何定义泛型类型，其中键类型使用 `comparable` 约束。
</details>
