## 快速预览

### 函数基础

```go
package main

import "fmt"

// 无参数无返回值
func greet() {
    fmt.Println("Hello, World!")
}

// 带参数和返回值
func add(a, b int) int {
    return a + b
}

// 多返回值（Go 特色）
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("division by zero")
    }
    return a / b, nil
}

func main() {
    greet()
    result := add(5, 3)
    fmt.Println(result)  // 8

    q, err := divide(10, 3)
    if err != nil {
        fmt.Println("错误:", err)
        return
    }
    fmt.Println("商:", q)
}
```

### 方法基础

```go
package main

import "fmt"

type Rectangle struct {
    width  float64
    height float64
}

// 值接收者方法
func (r Rectangle) Area() float64 {
    return r.width * r.height
}

// 指针接收者方法
func (r *Rectangle) Scale(factor float64) {
    r.width *= factor
    r.height *= factor
}

func main() {
    rect := Rectangle{width: 3, height: 4}
    fmt.Println("面积:", rect.Area())  // 12

    rect.Scale(2)
    fmt.Printf("缩放后: %.1f x %.1f\n", rect.width, rect.height)
    // 输出: 缩放后: 6.0 x 8.0
}
```

## 核心概念速览

### 函数 vs 方法

| 特性 | 函数 | 方法 |
|-----|------|------|
| 定义 | `func name()` | `func (recv T) name()` |
| 接收者 | 无 | 值或指针 |
| 调用 | `name()` | `recv.name()` |
| 用途 | 独立功能 | 类型行为 |

### 关键特性

<Badge text="多返回值" type="info" />
Go 的多返回值常用于错误处理：

```go
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("division by zero")
    }
    return a / b, nil
}
```

<Badge text="闭包" type="warning" />
函数可以捕获外部变量：

```go
func makeCounter() func() int {
    count := 0
    return func() int {
        count++
        return count
    }
}
```

<Badge text="Defer" type="warning" />
确保函数返回前执行代码：

```go
file, err := os.Open(path)
if err != nil {
    return err
}
defer file.Close()  // 函数返回前自动关闭
```

## 常见错误

<Callout type="danger" title="注意！">
  避免以下常见错误：

  1. **忽略错误返回值** - 始终检查 error
  2. **闭包循环变量** - Go 1.22+ 已修复
  3. **Defer 在循环中** - defer 不会立即执行
  4. **指针接收者混用** - 保持一致性
  5. **Panic 用于正常错误** - 应该返回 error
</Callout>

1. **编写函数判断一个数是否为质数**

<details>
<summary>查看答案</summary>

```go
package main

import "fmt"

// 方法一：基础实现
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    if n <= 3 {
        return true
    }
    if n%2 == 0 || n%3 == 0 {
        return false
    }

    for i := 5; i*i <= n; i += 6 {
        if n%i == 0 || n%(i+2) == 0 {
            return false
        }
    }
    return true
}

// 方法二：优化实现（只检查到平方根）
func isPrimeOptimized(n int) bool {
    if n <= 1 {
        return false
    }
    if n == 2 {
        return true
    }
    if n%2 == 0 {
        return false
    }

    for i := 3; i*i <= n; i += 2 {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    numbers := []int{2, 3, 4, 5, 17, 18, 19, 20}

    fmt.Println("质数判断结果:")
    for _, n := range numbers {
        if isPrime(n) {
            fmt.Printf("%d 是质数\n", n)
        } else {
            fmt.Printf("%d 不是质数\n", n)
        }
    }
}
```

**解释**：优化版本跳过偶数，只检查到平方根，性能更优。
</details>

<Badge text="中级" type="info" />
2. **创建一个 BankAccount 类型**，使用方法实现存款、取款、查询余额

<details>
<summary>查看答案</summary>

```go
package main

import "fmt"

// BankAccount 银行账户类型
type BankAccount struct {
    owner   string
    balance float64
}

// NewBankAccount 创建新账户
func NewBankAccount(owner string, initialBalance float64) *BankAccount {
    if initialBalance < 0 {
        initialBalance = 0
    }
    return &BankAccount{
        owner:   owner,
        balance: initialBalance,
    }
}

// Deposit 存款
func (ba *BankAccount) Deposit(amount float64) error {
    if amount <= 0 {
        return fmt.Errorf("存款金额必须大于0")
    }
    ba.balance += amount
    fmt.Printf("存款 %.2f 成功\n", amount)
    return nil
}

// Withdraw 取款
func (ba *BankAccount) Withdraw(amount float64) error {
    if amount <= 0 {
        return fmt.Errorf("取款金额必须大于0")
    }
    if amount > ba.balance {
        return fmt.Errorf("余额不足，当前余额: %.2f", ba.balance)
    }
    ba.balance -= amount
    fmt.Printf("取款 %.2f 成功\n", amount)
    return nil
}

// Balance 查询余额
func (ba *BankAccount) Balance() float64 {
    return ba.balance
}

// String 实现Stringer接口
func (ba BankAccount) String() string {
    return fmt.Sprintf("账户持有人: %s, 余额: %.2f", ba.owner, ba.balance)
}

func main() {
    account := NewBankAccount("张三", 1000.00)
    fmt.Println(account)

    // 存款
    err := account.Deposit(500.00)
    if err != nil {
        fmt.Println("错误:", err)
    }
    fmt.Printf("当前余额: %.2f\n", account.Balance())

    // 取款
    err = account.Withdraw(200.00)
    if err != nil {
        fmt.Println("错误:", err)
    }
    fmt.Printf("当前余额: %.2f\n", account.Balance())

    // 尝试取款超过余额
    err = account.Withdraw(2000.00)
    if err != nil {
        fmt.Println("错误:", err)
    }

    fmt.Println(account)
}
```

**解释**：展示了类型定义、构造函数、指针接收者方法、错误处理和Stringer接口实现。
</details>

3. **实现一个带缓存的函数**，使用闭包缓存计算结果

<details>
<summary>查看答案</summary>

```go
package main

import "fmt"

// 创建带缓存的斐波那契函数
func memoizedFibonacci() func(int) int {
    cache := make(map[int]int)

    var fib func(int) int
    fib = func(n int) int {
        if n <= 1 {
            return n
        }

        // 检查缓存
        if result, ok := cache[n]; ok {
            return result
        }

        // 计算并缓存
        result := fib(n-1) + fib(n-2)
        cache[n] = result
        return result
    }

    return fib
}

// 创建带缓存和统计的函数
func memoizeWithStats(fn func(int) int) func(int) int {
    cache := make(map[int]int)
    hits := 0
    misses := 0

    return func(n int) int {
        if result, ok := cache[n]; ok {
            hits++
            fmt.Printf("缓存命中! (命中: %d, 未命中: %d)\n", hits, misses)
            return result
        }

        misses++
        result := fn(n)
        cache[n] = result
        fmt.Printf("缓存未命中 (命中: %d, 未命中: %d)\n", hits, misses)
        return result
    }
}

// 创建通用的记忆化函数
func Memoize[T comparable, R any](fn func(T) R) func(T) R {
    cache := make(map[T]R)

    return func(input T) R {
        if result, ok := cache[input]; ok {
            return result
        }

        result := fn(input)
        cache[input] = result
        return result
    }
}

// 昂贵的计算函数
func expensiveCalculation(n int) int {
    fmt.Printf("正在计算 %d...\n", n)
    result := 0
    for i := 1; i <= n; i++ {
        result += i
    }
    return result
}

func main() {
    fmt.Println("=== 斐波那契记忆化 ===")
    fib := memoizedFibonacci()

    fmt.Println("fib(40) =", fib(40))   // 首次计算，较慢
    fmt.Println("fib(40) =", fib(40))   // 从缓存读取，极快
    fmt.Println("fib(35) =", fib(35))   // 首次计算
    fmt.Println("fib(40) =", fib(40))   // 从缓存读取

    fmt.Println("\n=== 带统计的记忆化 ===")
    memoCalc := memoizeWithStats(expensiveCalculation)

    memoCalc(100)  // 未命中
    memoCalc(100)  // 命中
    memoCalc(200)  // 未命中
    memoCalc(100)  // 命中
    memoCalc(200)  // 命中

    fmt.Println("\n=== 通用记忆化函数 ===")
    // 阶乘函数
    factorial := func(n int) int {
        if n <= 1 {
            return 1
        }
        return n * factorial(n-1)
    }

    memoFactorial := Memoize(factorial)

    fmt.Println("5! =", memoFactorial(5))   // 首次计算
    fmt.Println("5! =", memoFactorial(5))   // 缓存
    fmt.Println("7! =", memoFactorial(7))   // 会重复计算一些子问题
    fmt.Println("5! =", memoFactorial(5))   // 缓存
}
```

**解释**：展示了闭包如何实现缓存（记忆化），避免重复计算提升性能。
</details>
