```go
// 定义接口
type UserRepository interface {
    GetByID(id int) (*User, error)
    Create(user *User) error
}

type UserService struct {
    repo UserRepository
    emailSender EmailSender
}

func NewUserService(repo UserRepository, emailSender EmailSender) *UserService {
    return &UserService{
        repo:        repo,
        emailSender: emailSender,
    }
}

// 单元测试：使用 mock
func TestUserService_CreateUser_Unit(t *testing.T) {
    mockRepo := new(MockRepository)
    mockEmail := new(MockEmailSender)

    mockRepo.On("Create", mock.AnythingOfType("*User")).Return(nil)
    mockEmail.On("SendWelcome", mock.AnythingOfType("*User")).Return(nil)

    service := NewUserService(mockRepo, mockEmail)
    err := service.CreateUser("Alice", "alice@example.com")

    assert.NoError(t, err)
    mockRepo.AssertExpectations(t)
    mockEmail.AssertExpectations(t)
}

// 集成测试：使用真实数据库
func TestUserService_CreateUser_Integration(t *testing.T) {
    db := setupTestDB(t)
    defer db.Close()

    repo := NewUserRepository(db)
    emailSender := new(MockEmailSender)

    service := NewUserService(repo, emailSender)
    user, err := service.CreateUser("Alice", "alice@example.com")

    assert.NoError(t, err)
    assert.NotZero(t, user.ID)

    // 验证数据库中的数据
    dbUser, err := repo.GetByID(user.ID)
    assert.NoError(t, err)
    assert.Equal(t, "Alice", dbUser.Name)
}
```

## 表驱动测试架构

### 通用测试结构

```go
// 定义通用的测试结构
type TestCase struct {
    Name    string
    Input   interface{}
    Want    interface{}
    WantErr bool
}

// 定义通用的测试执行器
func RunTests(t *testing.T, tests []TestCase, testFunc func(TestCase) (interface{}, error)) {
    for _, tt := range tests {
        t.Run(tt.Name, func(t *testing.T) {
            got, err := testFunc(tt)

            if (err != nil) != tt.WantErr {
                t.Errorf("error = %v, wantErr %v", err, tt.WantErr)
                return
            }

            if !tt.WantErr && !reflect.DeepEqual(got, tt.Want) {
                t.Errorf("got %v, want %v", got, tt.Want)
            }
        })
    }
}

// 使用通用测试执行器
func TestParseInt(t *testing.T) {
    tests := []TestCase{
        {"valid number", "123", 123, false},
        {"invalid", "abc", nil, true},
        {"empty", "", nil, true},
    }

    RunTests(t, tests, func(tt TestCase) (interface{}, error) {
        return strconv.ParseInt(tt.Input.(string), 10, 64)
    })
}
```

```go
package auth_test
```
